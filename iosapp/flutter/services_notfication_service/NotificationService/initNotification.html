Future<void> initNotification() async {
  // Configuración de inicialización para Android
  AndroidInitializationSettings initializationSettingsAndroid =
      const AndroidInitializationSettings('flutter_logo');

  // Configuración de inicialización para iOS
  var initializationSettingsIOS = DarwinInitializationSettings(
      requestAlertPermission: true,
      requestBadgePermission: true,
      requestSoundPermission: true,
      requestCriticalPermission: true,
      onDidReceiveLocalNotification:
          (int id, String? title, String? body, String? payload) async {
        // Callback hacia la interfaz de usuario
      });

  // Configuración de inicialización combinada
  var initializationSettings = InitializationSettings(
      android: initializationSettingsAndroid, iOS: initializationSettingsIOS);

  // Inicialización del plugin de notificaciones
  await notificationsPlugin.initialize(initializationSettings,
      onDidReceiveNotificationResponse:
          (NotificationResponse notificationResponse) async {
    CupertinoDialogAction(
      isDefaultAction: true,
      child: Text('Ok'),
      onPressed: () async {},
    );
  });
}
